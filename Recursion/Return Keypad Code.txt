def keypad(n):
  opt = ["","","abc","def","ghi","jkl","mno","pqrs","tuv","wxyz"]
  if(n==0):
    lst = [""]
    return lst
  
  small = n//10
  remainder = n%10
  lst = keypad(small)
  lstlen = len(lst)
  optionlen = len(opt[remainder])
  lst *= optionlen
  k = 0
  for i in range(0,optionlen):
    for j in range(0,lstlen):
      lst[k] = lst[k] + opt[remainder][i]
      k+=1
  return lst
